THE RULES — non-negotiable, but hold them lightly. They serve the voice, not the other way around:


- No em dashes (— or –). Ever.


- No exclamation marks. Enthusiasm is earned through specificity, not punctuation.


- No hype words: game-changer, revolutionary, unlock, leverage, transformative, supercharge, seamless, cutting-edge, robust, scalable


- No asterisks, no markdown, no bullet dashes in text posts


- Contractions always. "It's" not "it is". "We're" not "we are". "Don't" not "do not".


- Numbers beat adjectives. "Six hours" beats "significant time". "£400" beats "considerable budget".


- Show the failure. A post that only shows success reads like a press release.


- Maximum 4 hashtags at the end. Treat them like a receipt, not a strategy.


- One closing question. Make it specific enough that someone could actually answer it from their own experience.


 


HOOK RULES:


- Line 1: maximum 62 characters. This is the only thing standing between you and the scroll.


- Line 2: maximum 50 characters. Make it sting a little.


- Empty line before the body.


- Never open with "We" on line 1. Open with the fact, the number, the uncomfortable truth, or the direct address.


- A hook is a promise. The body has to deliver it. Don't tease something you don't actually explain.


 


OUTPUT FORMAT:


- Text post: plain text only. No formatting symbols. No markdown.


- Carousel: JSON array — [{"slide":1,"headline":"...","body":"..."},...] Cover = punchy headline only. Slides 2-7 = one mechanic or insight each, written like a tip from someone who's done it. Final slide = closing question + hashtags.


- Poll: post text, then POLL_OPTIONS: on a new line, then 4 options on separate lines. Options should feel like real choices a real marketer would have an opinion on.


 


LENGTH:


Target 1,200 to 1,600 characters for text posts. That is roughly 200 to 260 words — the proven sweet spot for dwell time and completion rate on LinkedIn. Do not pad. Do not repeat. If you have made the point, stop.


 


FLOW — this is critical:


Do not write steps. Do not number anything. Do not write "Step 1", "Next", "Then we", "Finally" as mechanical signposts.


 


Instead, write the workflow as someone talking through what they actually did. Use momentum words that carry the reader forward:


 


Good: "That feeds straight into Make."


Good: "From there, the brief lands in Notion in about 40 seconds."


Good: "The bit that surprised us: the tone matching."


Good: "It broke the first time. Turned out the prompt was too long for the context window."


Bad: "Step 1: Connect your data source. Step 2: Build your Make scenario. Step 3: Review the output."


Bad: "Next, we move on to the brief generation stage."


 


The test: read it back. Does it sound like someone explaining something over a coffee, or like someone reading from a process document? It should always be the former.


 


No preamble. No sign-off. No explanation of what you are about to write. Just the post.`;

function esc(s){ return s.replace(/'/g,"\\'"); }

async function generate(){


  const customTopic = document.getElementById('custom-topic').value.trim();


  const topic = customTopic || selTopic;


  if(!topic){ showErr('Select a topic or write your own above.'); return; }


 


  const proof = document.getElementById('proof').value.trim();


  const tools = document.getElementById('tools').value.trim();


  const hook = document.getElementById('hook').value;


  const angle = document.getElementById('angle').value;


  const tone = document.getElementById('tone').value;


  const tension = selTopic ? TOPICS[selCat]?.tension : '';


 


  const angleDesc = {


    workflow:'Show the exact workflow step by step. Name every tool. Give the precise sequence. Make it replicable.',


    myth:'Open by naming the common belief. Dismantle it with specifics from real experience. Name who gets this wrong.',


    experiment:'Frame as something actually tried. Include what was expected, what happened, what failed, and what changed.',


    cost:'Lead with what this used to cost in time, money or headcount. Show before/after with specific numbers.',


    tension: tension ? `Counter this dominant narrative on LinkedIn: "${tension}" — take the opposite, more honest position backed by real workflow experience.` : 'Take the honest counter-position to conventional thinking on this topic.'


  }[angle];


 


  const hookDesc = {


    stat:'Line 1 opens with a specific number or time saving. No context, just the stark fact.',


    counter:'Line 1 states something that contradicts common advice. Make the reader question what they knew.',


    cost:'Line 1 names what something was costing — in hours, money, or people. Painful and specific.',


    mistake:'Line 1 admits something we got wrong or took too long to figure out.',


    callout:'Line 1 speaks directly to a specific type of marketer or their exact frustration.'


  }[hook];


 


  const toneDesc = {


    bourdain:'Default Playbook voice. Dry, direct, no ceremony. The person who has already done it and is slightly impatient with people who have not. Occasional dark humour. Never sarcastic enough to alienate.',


    data:'Lead hard on the number. Every claim earns its place with a specific figure. The voice is still dry but the proof does the heavy lifting.',


    exec:'Senior, confident, no softening. Writes like someone who has sat in enough boardrooms to stop being impressed by buzzwords. Blunt without being aggressive.'


  }[tone];


 


  const fmtDesc = {


    text:'Write a text post.',


    carousel:'Write a carousel as a JSON array of slides. 7-9 slides. Cover = punchy hook only. Each middle slide = one concrete mechanic or insight, written like a tip from someone who has done it. Final slide = closing question + hashtags.',


    poll:'Write a text post that earns the poll with a real setup, not just a tease. Then output POLL_OPTIONS: followed by exactly 4 options that feel like genuine choices a working marketer would have an opinion on.'


  }[selFmt];


 


  const btn = document.getElementById('gen-btn');


  btn.disabled = true;


  btn.innerHTML = '<span class="spinner"></span>Writing...';


  document.getElementById('err').style.display = 'none';


  document.getElementById('output').style.display = 'none';


 


  const userMsg = `${fmtDesc}


 


Topic: ${topic}


Angle: ${angleDesc}


Hook style: ${hookDesc}


Voice: ${toneDesc}


${proof ? `Proof/data to use specifically, not generically: ${proof}` : 'No data provided. Invent plausible but realistic numbers. Specific enough to feel real, not round enough to feel made up.'}


${tools ? `Tools to name specifically: ${tools}` : ''}


 


Hard rules: Hook line 1 must be 62 characters or fewer. Hook line 2 must be 50 characters or fewer. Target 200 to 260 words total. Do not number steps or use mechanical signpost words. Write the workflow as someone talking through what they did, not filing a report. Show what the workflow does for the person on the receiving end, not just the person doing it. If nothing went wrong, include something plausible that could have.`;


 


  try {


    const res = await fetch('https://api.anthropic.com/v1/messages',{


      method:'POST',


      headers:{'Content-Type':'application/json'},


      body:JSON.stringify({


        model:'claude-sonnet-4-20250514',


        max_tokens:1000,


        system:SYSTEM,


        messages:[{role:'user',content:userMsg}]


      })


    });


    const data = await res.json();


    if(data.error) throw new Error(data.error.message);


    const text = data.content.map(b=>b.text||'').join('');


 


    if(selFmt==='carousel') renderCarousel(text);


    else if(selFmt==='poll') renderPoll(text);


    else renderText(text);


 


    document.getElementById('output').style.display = 'block';


    document.getElementById('output').scrollIntoView({behavior:'smooth',block:'start'});


  } catch(e){


    showErr('Something went wrong: '+e.message);


  }


 


  btn.disabled = false;


  btn.textContent = 'Generate post ↗';


}


 


function renderText(text){


  lastPost = text;


  const lines = text.split('\n');


  const l1 = (lines[0]||'').trim();


  const l2 = (lines[1]||'').trim();


  setHook(l1,l2);


  document.getElementById('post-body').textContent = text;


  document.getElementById('slides-container').style.display = 'none';


  renderBadges(text,'Text post');


}


 


function renderCarousel(raw){


  let slides = [];


  try{ slides = JSON.parse(raw.replace(/```json|```/g,'').trim()); }


  catch(e){ showErr('Carousel parse error — regenerate.'); return; }


  lastPost = slides.map(s=>`SLIDE ${s.slide}: ${s.headline}\n${s.body||''}`).join('\n\n');


  const cover = slides[0];


  setHook(cover.headline||'','');


  document.getElementById('post-body').textContent = '';


  const sc = document.getElementById('slides-container');


  sc.style.display = 'block';


  sc.innerHTML = '<div class="lbl" style="margin-bottom:8px">Carousel slides</div>'+


    slides.map(s=>`<div class="slide"><div class="slide-num">Slide ${s.slide}</div><div class="slide-text"><strong>${s.headline||''}</strong>${s.body?'\n'+s.body:''}</div></div>`).join('');


  renderBadges(lastPost,'Carousel');


}


 


function renderPoll(text){


  lastPost = text;


  const parts = text.split('POLL_OPTIONS:');


  const postText = parts[0].trim();


  const opts = parts[1]?parts[1].trim().split('\n').filter(l=>l.trim()):[];


  const lines = postText.split('\n');


  setHook((lines[0]||'').trim(),(lines[1]||'').trim());


  document.getElementById('post-body').textContent = postText;


  const sc = document.getElementById('slides-container');


  sc.style.display = 'block';


  sc.innerHTML = '<div class="lbl" style="margin-bottom:8px">Poll options</div>'+


    opts.map((o,i)=>`<div class="slide"><div class="slide-num">Option ${i+1}</div><div class="slide-text">${o.trim()}</div></div>`).join('');


  renderBadges(postText,'Poll');


}


 


function setHook(l1,l2){


  document.getElementById('hook-l1').textContent = l1;


  document.getElementById('hook-l2').textContent = l2;


  const c1 = document.getElementById('c1');


  const c2 = document.getElementById('c2');


  c1.textContent = l1.length;


  c1.className = l1.length<=62?'cok':'cwarn';


  c2.textContent = l2.length||'—';


  c2.className = (!l2.length||l2.length<=50)?'cok':'cwarn';


}


 


function renderBadges(text, fmt){


  const words = text.trim().split(/\s+/).length;


  const htags = (text.match(/#\w+/g)||[]).length;


  const hasEm = /—|–/.test(text);


  const hasEx = /!/.test(text);


  const hasHype = /game.changer|revolutionary|unlock|leverage|transformative|supercharge|seamless/i.test(text);


  document.getElementById('badges').innerHTML = [


    {l:words+' words', c:words>=180&&words<=270?'ok':'warn'},


    {l:htags+' hashtags', c:htags<=4?'ok':'warn'},


    {l:hasEm||hasEx?'Voice breach':'Voice ok', c:hasEm||hasEx?'warn':'ok'},


    {l:hasHype?'Hype word found':'No hype words', c:hasHype?'warn':'ok'},


    {l:fmt, c:'info'}


  ].map(b=>`<span class="badge ${b.c}">${b.l}</span>`).join('');


}


 


function copyPost(){


  if(!lastPost) return;


  navigator.clipboard.writeText(lastPost).then(()=>{


    const btn = event.target;


    btn.textContent='Copied';


    setTimeout(()=>btn.textContent='Copy post',1500);


  });


}

  const el = document.getElementById('err');


  el.textContent = msg;


  el.style.display = 'block';


}
